```python
def build_option_pmf(
    opt: str,
    item: ItemBase,
    weights: JobWeights,
    stage_probs: Dict[int, float],
    armor_table: Dict[int, Dict[str, List[Optional[int]]]],
    is_weapon: bool
) -> Optional[Dict[int, float]]:
    pmf: Dict[int, float] = {}
    total_p = 0.0

    for s, p in stage_probs.items():
        sc = score_from_option(opt, item, s, weights, armor_table, is_weapon)
        if sc is None:
            # ì´ ì˜µì…˜ì´ í•´ë‹¹ ë‹¨ê³„ê°’ì„ ì •ì˜í•˜ì§€ ì•Šìœ¼ë©´, ì´ ì¡°í•© ìì²´ë¥¼ "ë¶ˆê°€ëŠ¥"ë¡œ ë³´ëŠ” ê²Œ ì•ˆì „í•¨
            return None
        pmf[sc] = pmf.get(sc, 0.0) + p

    # stage_probsëŠ” ë³´í†µ í•©ì´ 1ì´ì§€ë§Œ í˜¹ì‹œ ëª¨ë¥´ë‹ˆ ì •ê·œí™”
    if total_p > 0 and abs(total_p - 1.0) > 1e-9:

            pmf[k] /= total_p
    return pmf

def build_option_pmfs(

    item: ItemBase,
    weights: JobWeights,
    stage_probs: Dict[int, float],
    armor_table: Dict[int, Dict[str, List[Optional[int]]]],
    is_weapon: bool
) -> List[Dict[int, float]]:
    pmfs: List[Dict[int, float]] = []
    for opt in option_pool:
        pmf = build_option_pmf(opt, item, weights, stage_probs, armor_table, is_weapon)
        if pmf is not None:
            pmfs.append(pmf)
    return pmfs
```

```python
def prob_at_least_T_fixed_k(option_pmfs: List[Dict[int, float]], k: int, T: int) -> float:
    N = len(option_pmfs)
    if k > N:
        return 0.0

    # dp[r][t] = ì•ì—ì„œ ì²˜ë¦¬í•œ ì˜µì…˜ë“¤ë¡œ rê°œë¥¼ ì„ íƒí–ˆì„ ë•Œ, ìº¡ ì ìˆ˜ tê°€ ë˜ëŠ” í™•ë¥  "í•©(ì¡°í•©ë“¤ ì „ì²´)"
    dp = [[0.0] * (T + 1) for _ in range(k + 1)]
    dp[0][0] = 1.0

    for i in range(N):
        pmf = option_pmfs[i]
        new = [row[:] for row in dp]  # "ì•ˆ ë½‘ëŠ” ê²½ìš°" ë°˜ì˜

        rmax = min(i + 1, k)
        for r in range(rmax, 0, -1):
            for t in range(T + 1):
                base = dp[r - 1][t]
                if base == 0.0:
                    continue
                for val, p in pmf.items():
                    nt = T if t + val >= T else t + val
                    new[r][nt] += base * p

        dp = new

    # ê· ë“± ì¡°í•© í‰ê· ìœ¼ë¡œ ë°”ê¾¸ê¸° ìœ„í•´ C(N,k)ë¡œ ë‚˜ëˆ”
    return dp[k][T] / math.comb(N, k)

def prob_at_least_T_other_equipment(option_pmfs: List[Dict[int, float]], T: int) -> float:
    # ë„ˆì˜ ê·œì¹™: ê·¸ì™¸ ì¥ë¹„ëŠ” k=1~4 ê· ë“±
    N = len(option_pmfs)
    ks = [k for k in (1, 2, 3, 4) if k <= N]
    if not ks:
        return 0.0

    return sum(prob_at_least_T_fixed_k(option_pmfs, k, T) for k in ks) / 4.0
```
